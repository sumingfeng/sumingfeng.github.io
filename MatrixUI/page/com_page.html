JS:

<script src=""../js/ui/su_pagination.js"></script>
<script src=""../js/ui/su_tableStyle.js"></script>
<script>
	//模拟数据
	var arr = [];
	for (var i = 0; i < 100; i++) {
		var date = new Date();
		arr.push({
			number: i + 1,
			_date: date.getHours() + ':' + date.getMinutes() + ':' + date.getSeconds()
		});
	}

	$(function() {
		//渲染分页
		$('#pagination').pagination(arr.length, {
			current_page: 0, //当前页
			items_per_page: 10, //每页显示的条目数
			num_display_entries: 5, //每5页展示、分页条目数
			callback: function(page_id) {
				//模拟ajax去后端读取页数，获取数据并渲染列表的过程
				$('#list-content').html('');
				var dataHTML = '';
				for (var i = 0; i < 10; i++) {
					dataHTML += '<tr><td>' + arr[page_id * 10 + i].number + '</td><td>' + arr[page_id * 10 + i]._date + '</td></tr>'
				}
				$('#list-content').append(dataHTML);

				$('.datatable').su_tableStyle({
					odd: "tr-odd"
				});
			},
			load_first_page: true,
			prev_text: '上一页',
			next_text: '下一页'
		});

	});
</script>
